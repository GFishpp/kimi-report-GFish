```bash
#!/bin/bash
# scripts/secrets-manager.sh

SECRETS_DIR="${HOME}/.openclaw/secrets"
KEY_FILE="${SECRETS_DIR}/master.key"

# 初始化密钥存储
init_secrets() {
    mkdir -p "${SECRETS_DIR}"
    chmod 700 "${SECRETS_DIR}"
    
    # 生成主密钥
    if [[ ! -f "${KEY_FILE}" ]]; then
        openssl rand -base64 32 > "${KEY_FILE}"
        chmod 600 "${KEY_FILE}"
    fi
}

# 加密存储密钥
store_secret() {
    local name="$1"
    local value="$2"
    
    echo "${value}" | openssl enc -aes-256-cbc -salt \
        -pass "file:${KEY_FILE}" \
        -out "${SECRETS_DIR}/${name}.enc"
    
    chmod 600 "${SECRETS_DIR}/${name}.enc"
}

# 读取密钥
get_secret() {
    local name="$1"
    
    openssl enc -aes-256-cbc -d \
        -pass "file:${KEY_FILE}" \
        -in "${SECRETS_DIR}/${name}.enc" 2>/dev/null
}

# 环境变量注入（临时）
inject_env() {
    local prefix="OPENCLAW_SECRET_"
    
    for enc_file in "${SECRETS_DIR}"/*.enc; do
        [[ -f "${enc_file}" ]] || continue
        
        local name=$(basename "${enc_file}" .enc)
        local value=$(get_secret "${name}")
        
        export "${prefix}${name}=${value}"
    done
}
```

```python
# lib/log_sanitizer.py
import re
from typing import Pattern, List

class LogSanitizer:
    """日志脱敏处理器"""
    
    # 敏感信息正则模式
    PATTERNS: List[tuple] = [
        # API Keys
        (re.compile(r'(api[_-]?key\s*[=:]\s*)["\']?[a-zA-Z0-9]{32,}["\']?', re.I), r'\1***REDACTED***'),
        # 密码
        (re.compile(r'(password\s*[=:]\s*)["\']?[^"\'\s]+["\']?', re.I), r'\1***REDACTED***'),
        # Token
        (re.compile(r'(token\s*[=:]\s*)["\']?[a-zA-Z0-9_-]+\.[a-zA-Z0-9_-]+\.[a-zA-Z0-9_-]+["\']?', re.I), r'\1***REDACTED***'),
        # 邮箱
        (re.compile(r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'), r'***EMAIL***'),
        # IP地址
        (re.compile(r'\b(?:\d{1,3}\.){3}\d{1,3}\b'), r'***IP***'),
        # 手机号
        (re.compile(r'\b1[3-9]\d{9}\b'), r'***PHONE***'),
        # 身份证号
        (re.compile(r'\b\d{17}[\dXx]|\d{15}\b'), r'***ID***'),
    ]
    
    @classmethod
    def sanitize(cls, text: str) -> str:
        """对文本进行脱敏处理"""
        for pattern, replacement in cls.PATTERNS:
            text = pattern.sub(replacement, text)
        return text
    
    @classmethod
    def sanitize_dict(cls, data: dict, sensitive_keys: List[str] = None) -> dict:
        """对字典中的敏感字段进行脱敏"""
        if sensitive_keys is None:
            sensitive_keys = ['password', 'token', 'secret', 'key', 'api_key', 'private_key']
        
        result = {}
        for key, value in data.items():
            if any(sk in key.lower() for sk in sensitive_keys):
                result[key] = '***REDACTED***'
            elif isinstance(value, dict):
                result[key] = cls.sanitize_dict(value, sensitive_keys)
            elif isinstance(value, str):
                result[key] = cls.sanitize(value)
            else:
                result[key] = value
        return result
```

### 2.1 上下文管理

#### 2.1.1 上下文窗口优化策略

```python
# lib/context_manager.py
from typing import List, Dict, Optional
import tiktoken

class ContextOptimizer:
    """上下文窗口优化管理器"""
    
    def __init__(self, max_tokens: int = 128000, model: str = "gpt-4"):
        self.max_tokens = max_tokens
        self.model = model
        self.encoding = tiktoken.encoding_for_model(model)
        
    def count_tokens(self, text: str) -> int:
        """计算文本的token数量"""
        return len(self.encoding.encode(text))
    
    def optimize_context(
        self,
        messages: List[Dict],
        priority_rules: Optional[Dict] = None
    ) -> List[Dict]:
        """
        优化上下文窗口
        
        策略：
        1. 保留系统消息
        2. 保留最近的对话
        3. 压缩或移除旧的历史
        4. 优先保留高优先级内容
        """
        if priority_rules is None:
            priority_rules = {
                'system': 100,      # 系统消息最高优先级
                'user_recent': 90,  # 最近用户消息
                'assistant_recent': 80,  # 最近助手回复
                'tool_result': 70,  # 工具执行结果
                'history': 50,      # 历史对话
            }
        
        # 计算当前token数
        total_tokens = sum(
            self.count_tokens(msg.get('content', ''))
            for msg in messages
        )
        
        if total_tokens <= self.max_tokens * 0.8:
            return messages
        
        # 按优先级排序
        prioritized = []
        for i, msg in enumerate(messages):
            role = msg.get('role', '')
            
            # 确定优先级
            if role == 'system':
                priority = priority_rules['system']
            elif i >= len(messages) - 4:
                priority = priority_rules.get(f'{role}_recent', 50)
            elif msg.get('tool_calls') or msg.get('tool_call_id'):
                priority = priority_rules['tool_result']
            else:
                priority = priority_rules['history']
            
            prioritized.append((priority, i, msg))
        
        # 按优先级降序排序
        prioritized.sort(key=lambda x: (-x[0], x[1]))
        
        # 选择消息直到接近限制
        selected = []
        current_tokens = 0
        
        for priority, original_index, msg in prioritized:
            msg_tokens = self.count_tokens(msg.get('content', ''))
            
            if current_tokens + msg_tokens <= self.max_tokens * 0.9:
                selected.append((original_index, msg))
                current_tokens += msg_tokens
        
        # 按原始顺序排序
        selected.sort(key=lambda x: x[0])
        
        return [msg for _, msg in selected]
    
    def summarize_old_context(
        self,
        messages: List[Dict],
        keep_recent: int = 6
    ) -> List[Dict]:
        """对旧上下文进行摘要"""
        if len(messages) <= keep_recent:
            return messages
        
        # 保留最近的消息
        recent = messages[-keep_recent:]
        old = messages[:-keep_recent]
        
        # 对旧消息生成摘要（实际应用中可调用LLM）
        old_content = '\n'.join([
            f"{msg.get('role')}: {msg.get('content', '')[:200]}..."
            for msg in old
        ])
        
        summary = {
            'role': 'system',
            'content': f'[历史对话摘要] 之前的对话包含 {len(old)} 条消息，主要涉及:\n{old_content[:500]}...'
        }
        
        return [summary] + recent
```

#### 2.1.2 上下文缓存策略

```python
# lib/context_cache.py
import hashlib
import json
from functools import lru_cache
from typing import Any, Optional
import redis

class ContextCache:
    """上下文缓存管理器"""
    
    def __init__(self, redis_url: Optional[str] = None):
        self.redis_client = redis.from_url(redis_url) if redis_url else None
        self.local_cache = {}
        self.ttl = 3600  # 1小时
    
    def _generate_key(self, context: Any) -> str:
        """生成缓存键"""
        context_str = json.dumps(context, sort_keys=True)
        return hashlib.sha256(context_str.encode()).hexdigest()[:32]
    
    def get(self, context: Any) -> Optional[Any]:
        """获取缓存"""
        key = self._generate_key(context)
        
        # 先查本地缓存
        if key in self.local_cache:
            return self.local_cache[key]
        
        # 再查Redis
        if self.redis_client:
            value = self.redis_client.get(key)
            if value:
                return json.loads(value)
        
        return None
    
    def set(self, context: Any, result: Any) -> None:
        """设置缓存"""
        key = self._generate_key(context)
        
        # 本地缓存
        self.local_cache[key] = result
        
        # Redis缓存
        if self.redis_client:
            self.redis_client.setex(
                key,
                self.ttl,
                json.dumps(result)
            )
    
    def invalidate(self, pattern: str = '*') -> int:
        """使缓存失效"""
        count = 0
        
        # 清空本地缓存
        if pattern == '*':
            self.local_cache.clear()
        else:
            keys_to_remove = [
                k for k in self.local_cache.keys()
                if pattern in k
            ]
            for k in keys_to_remove:
                del self.local_cache[k]
                count += 1
        
        # 清空Redis缓存
        if self.redis_client:
            for key in self.redis_client.scan_iter(match=pattern):
                self.redis_client.delete(key)
                count += 1
        
        return count
```

```python
# lib/concurrency_controller.py
import asyncio
from asyncio import Semaphore
from typing import Dict, Optional
from dataclasses import dataclass
import time

@dataclass
class RateLimiter:
    """速率限制器"""
    max_requests: int
    window_seconds: int = 60
    
    def __post_init__(self):
        self.requests = []
        self._lock = asyncio.Lock()
    
    async def acquire(self) -> bool:
        async with self._lock:
            now = time.time()
            # 清理过期请求
            self.requests = [
                req_time for req_time in self.requests
                if now - req_time < self.window_seconds
            ]
            
            if len(self.requests) < self.max_requests:
                self.requests.append(now)
                return True
            return False
    
    async def wait_for_slot(self) -> None:
        while not await self.acquire():
            await asyncio.sleep(0.1)

class ConcurrencyController:
    """并发控制器"""
    
    def __init__(self, config: Dict):
        self.config = config
        self.semaphores: Dict[str, Semaphore] = {}
        self.rate_limiters: Dict[str, RateLimiter] = {}
        self._init_limiters()
    
    def _init_limiters(self):
        """初始化限制器"""
        # 按工具类型初始化
        for tool, settings in self.config.get('by_tool', {}).items():
            max_concurrent = settings.get('max_concurrent', 5)
            self.semaphores[tool] = Semaphore(max_concurrent)
            
            rate_limit = settings.get('rate_limit_per_minute')
            if rate_limit:
                self.rate_limiters[tool] = RateLimiter(rate_limit)
    
    async def execute_with_limit(
        self,
        tool_name: str,
        coro,
        user_tier: str = 'default'
    ):
        """在并发限制下执行"""
        semaphore = self.semaphores.get(tool_name)
        rate_limiter = self.rate_limiters.get(tool_name)
        
        # 等待速率限制
        if rate_limiter:
            await rate_limiter.wait_for_slot()
        
        # 等待并发限制
        if semaphore:
            async with semaphore:
                return await coro
        else:
            return await coro

# 使用示例
controller = ConcurrencyController({
    'by_tool': {
        'web_search': {
            'max_concurrent': 10,
            'rate_limit_per_minute': 60
        }
    }
})

async def search_with_limit(query: str):
    return await controller.execute_with_limit(
        'web_search',
        web_search(query),
        user_tier='premium'
    )
```

```python
            memory_percent=mem.percent,
            memory_used_mb=mem.used / 1024 / 1024,
            disk_percent=disk.percent,
            open_files=len(self.process.open_files()),
            connections=len(self.process.connections()),
            thread_count=self.process.num_threads()
        )
    
    async def monitor_loop(self, interval: float = 5.0):
        """监控循环"""
        self._running = True
        
        while self._running:
            usage = self.get_current_usage()
            self.history.append(usage)
            
            # 限制历史记录大小
            if len(self.history) > self.max_history:
                self.history = self.history[-self.max_history:]
            
            # 检查阈值
            alerts = []
            if usage.cpu_percent > self.cpu_threshold:
                alerts.append(f"CPU使用率: {usage.cpu_percent:.1f}%")
            if usage.memory_percent > self.memory_threshold:
                alerts.append(f"内存使用率: {usage.memory_percent:.1f}%")
            if usage.disk_percent > self.disk_threshold:
                alerts.append(f"磁盘使用率: {usage.disk_percent:.1f}%")
            
            if alerts:
                for callback in self._callbacks:
                    try:
                        callback(alerts, usage)
                    except Exception as e:
                        print(f"回调执行失败: {e}")
            
            await asyncio.sleep(interval)
    
    def stop(self):
        """停止监控"""
        self._running = False
    
    def get_stats(self) -> dict:
        """获取统计信息"""
        if not self.history:
            return {}
        
        cpu_values = [h.cpu_percent for h in self.history]
        mem_values = [h.memory_percent for h in self.history]
        
        return {
            'samples': len(self.history),
            'cpu': {
                'current': cpu_values[-1],
                'avg': sum(cpu_values) / len(cpu_values),
                'max': max(cpu_values),
                'min': min(cpu_values)
            },
            'memory': {
                'current': mem_values[-1],
                'avg': sum(mem_values) / len(mem_values),
                'max': max(mem_values),
                'min': min(mem_values)
            }
        }
```

```bash
        local mem_usage=$(ps -p $$ -o %mem= | tr -d ' ')

            # 触发垃圾回收或重启
        fi
        
        sleep 5
    done
}
```

```python
#!/usr/bin/env python3
# scripts/log_analyzer.py

import re
import json
from datetime import datetime, timedelta
from collections import defaultdict
from pathlib import Path
from typing import Dict, List, Optional
import argparse

class LogAnalyzer:
    """日志分析器"""
    
    LOG_PATTERN = re.compile(
        r'^(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) '
        r'\[(\w+)\] '
        r'(\S+): '
        r'(.+)$'
    )
    
    def __init__(self, log_path: str):
        self.log_path = Path(log_path)
        self.entries: List[Dict] = []
        self.errors: List[Dict] = []
        self.warnings: List[Dict] = []
    
    def parse(self) -> 'LogAnalyzer':
        """解析日志文件"""
        if not self.log_path.exists():
            raise FileNotFoundError(f"日志文件不存在: {self.log_path}")
        
        with open(self.log_path, 'r', encoding='utf-8') as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                
                match = self.LOG_PATTERN.match(line)
                if match:
                    entry = {
                        'timestamp': datetime.strptime(
                            match.group(1), '%Y-%m-%d %H:%M:%S'
                        ),
                        'level': match.group(2),
                        'logger': match.group(3),
                        'message': match.group(4)
                    }
                    self.entries.append(entry)
                    
                    if entry['level'] == 'ERROR':
                        self.errors.append(entry)
                    elif entry['level'] == 'WARNING':
                        self.warnings.append(entry)
        
        return self
    
    def analyze(self) -> Dict:
        """分析日志"""
        if not self.entries:
            return {}
        
        # 时间范围
        start_time = min(e['timestamp'] for e in self.entries)
        end_time = max(e['timestamp'] for e in self.entries)
        duration = end_time - start_time
        
        # 级别统计
        level_counts = defaultdict(int)
        logger_counts = defaultdict(int)
        error_patterns = defaultdict(int)
        
        for entry in self.entries:
            level_counts[entry['level']] += 1
            logger_counts[entry['logger']] += 1
            
            if entry['level'] == 'ERROR':
                # 提取错误模式
                error_key = self._extract_error_pattern(entry['message'])
                error_patterns[error_key] += 1
        
        return {
            'summary': {
                'total_entries': len(self.entries),
                'time_range': {
                    'start': start_time.isoformat(),
                    'end': end_time.isoformat(),
                    'duration_hours': duration.total_seconds() / 3600
                },
                'error_count': len(self.errors),
                'warning_count': len(self.warnings)
            },
            'level_distribution': dict(level_counts),
            'logger_distribution': dict(logger_counts),
            'top_errors': dict(sorted(
                error_patterns.items(),
                key=lambda x: x[1],
                reverse=True
            )[:10])
        }
    
    def _extract_error_pattern(self, message: str) -> str:
        """提取错误模式"""
        # 移除具体的变量值，保留错误结构
        patterns = [
            (r"'[^']+'", "'***'"),
            (r'"[^"]+"', '"***"'),
            (r'\b\d+\b', '###'),
            (r'0x[0-9a-fA-F]+', '0x***'),
        ]
        
        result = message
        for pattern, replacement in patterns:
            result = re.sub(pattern, replacement, result)
        
        return result[:100]
    
    def generate_report(self, output_path: Optional[str] = None) -> str:
        """生成报告"""
        analysis = self.analyze()
        
        report_lines = [
            "=" * 60,
            "OpenClaw 日志分析报告",
            "=" * 60,
            "",
            "## 概览",
            f"- 总条目数: {analysis['summary']['total_entries']}",
            f"- 错误数: {analysis['summary']['error_count']}",
            f"- 警告数: {analysis['summary']['warning_count']}",
            f"- 时间范围: {analysis['summary']['time_range']['start']} ~ {analysis['summary']['time_range']['end']}",
            "",
            "## 日志级别分布",
        ]
        
        for level, count in analysis['level_distribution'].items():
            report_lines.append(f"- {level}: {count}")
        
        report_lines.extend([
            "",
            "## 常见错误模式 (Top 10)",
        ])
        
        for error, count in list(analysis['top_errors'].items())[:10]:
            report_lines.append(f"- [{count}次] {error}")
        
        report_lines.extend([
            "",
            "## 详细错误列表",
            "-" * 60,
        ])
        
        for error in self.errors[-20:]:  # 最近20个错误
            report_lines.append(
                f"[{error['timestamp']}] {error['logger']}: {error['message']}"
            )
        
        report = '\n'.join(report_lines)
        
        if output_path:
            Path(output_path).write_text(report, encoding='utf-8')
        
        return report

def main():
    parser = argparse.ArgumentParser(description='OpenClaw 日志分析工具')
    parser.add_argument('log_file', help='日志文件路径')
    parser.add_argument('-o', '--output', help='输出报告路径')
    parser.add_argument('-j', '--json', action='store_true', help='输出JSON格式')
    
    args = parser.parse_args()
    
    analyzer = LogAnalyzer(args.log_file).parse()
    
    if args.json:
        analysis = analyzer.analyze()
        print(json.dumps(analysis, indent=2, default=str))
    else:
        report = analyzer.generate_report(args.output)
        print(report)

if __name__ == '__main__':
    main()
```

```python
#!/usr/bin/env python3
# scripts/debug_tools.py

import asyncio
import functools
import time
import tracemalloc
import linecache
import sys
from typing import Any, Callable
from contextlib import contextmanager

class DebugTools:
    """调试工具集"""
    
    @staticmethod
    def trace_calls(func: Callable) -> Callable:
        """函数调用追踪装饰器"""
        @functools.wraps(func)
        async def async_wrapper(*args, **kwargs):
            print(f"[TRACE] Calling {func.__name__} with args={args}, kwargs={kwargs}")
            start = time.time()
            try:
                result = await func(*args, **kwargs)
                elapsed = time.time() - start
                print(f"[TRACE] {func.__name__} completed in {elapsed:.3f}s")
                return result
            except Exception as e:
                elapsed = time.time() - start
                print(f"[TRACE] {func.__name__} failed after {elapsed:.3f}s: {e}")
                raise
        
        @functools.wraps(func)
        def sync_wrapper(*args, **kwargs):
            print(f"[TRACE] Calling {func.__name__} with args={args}, kwargs={kwargs}")
            start = time.time()
            try:
                result = func(*args, **kwargs)
                elapsed = time.time() - start
                print(f"[TRACE] {func.__name__} completed in {elapsed:.3f}s")
                return result
            except Exception as e:
                elapsed = time.time() - start
                print(f"[TRACE] {func.__name__} failed after {elapsed:.3f}s: {e}")
                raise
        
        return async_wrapper if asyncio.iscoroutinefunction(func) else sync_wrapper
    
    @staticmethod
    @contextmanager
    def measure_time(operation_name: str):
        """测量代码块执行时间"""
        start = time.perf_counter()
        try:
            yield
        finally:
            elapsed = time.perf_counter() - start
            print(f"[TIMING] {operation_name}: {elapsed:.4f}s")
    
    @staticmethod
    @contextmanager
    def memory_tracker(top_n: int = 10):
        """内存使用追踪"""
        tracemalloc.start()
        try:
            yield
        finally:
            snapshot = tracemalloc.take_snapshot()
            top_stats = snapshot.statistics('lineno')
            
            print(f"\n[MEMORY] Top {top_n} memory allocations:")
            for stat in top_stats[:top_n]:
                frame = stat.traceback[0]
                filename = frame.filename
                lineno = frame.lineno
                print(f"  {filename}:{lineno} - {stat.size / 1024:.1f} KiB")
            
            tracemalloc.stop()
    
    @staticmethod
    def inspect_object(obj: Any, depth: int = 2) -> None:
        """检查对象详情"""
        print(f"\n[INSPECT] Object: {type(obj).__name__}")
        print(f"  ID: {id(obj)}")
        print(f"  Size: {sys.getsizeof(obj)} bytes")
        print(f"  Attributes:")
        
        for attr in dir(obj):
            if not attr.startswith('_'):
                try:
                    value = getattr(obj, attr)
                    if not callable(value):
                        value_str = repr(value)[:50]
                        print(f"    {attr}: {value_str}")
                except Exception as e:
                    print(f"    {attr}: <error: {e}>")
```

#### 4.1.2 工作流引擎

```python
# lib/workflow_engine.py
import asyncio
import yaml
from typing import Dict, List, Any, Optional
from dataclasses import dataclass, field
from enum import Enum
import json
from datetime import datetime

class TaskStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    SUCCESS = "success"
    FAILED = "failed"
    SKIPPED = "skipped"

@dataclass
class Task:
    id: str
    name: str
    type: str
    config: Dict[str, Any]
    depends_on: List[str] = field(default_factory=list)
    retries: int = 0
    status: TaskStatus = TaskStatus.PENDING
    output: Any = None
    error: Optional[str] = None
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None

@dataclass
class Workflow:
    name: str
    description: str
    version: str
    tasks: Dict[str, Task]
    env: Dict[str, str]
    on_error: List[Dict]

class WorkflowEngine:
    """工作流引擎"""
    
    def __init__(self):
        self.workflows: Dict[str, Workflow] = {}
        self.task_handlers: Dict[str, callable] = {
            'web_fetch': self._handle_web_fetch,
            'exec': self._handle_exec,
            'message': self._handle_message,
        }
    
    def load_workflow(self, path: str) -> Workflow:
        """加载工作流定义"""
        with open(path, 'r') as f:
            data = yaml.safe_load(f)
        
        workflow_data = data['workflow']
        
        tasks = {}
        for task_data in workflow_data.get('tasks', []):
            task = Task(
                id=task_data['id'],
                name=task_data['name'],
                type=task_data['type'],
                config=task_data.get('config', {}),
                depends_on=task_data.get('depends_on', []),
                retries=int(task_data.get('retries', 0))
            )
            tasks[task.id] = task
        
        workflow = Workflow(
            name=workflow_data['name'],
            description=workflow_data.get('description', ''),
            version=workflow_data.get('version', '1.0'),
            tasks=tasks,
            env=workflow_data.get('env', {}),
            on_error=workflow_data.get('on_error', [])
        )
        
        self.workflows[workflow.name] = workflow
        return workflow
    
    async def execute(self, workflow_name: str, context: Dict = None) -> Dict:
        """执行工作流"""
        workflow = self.workflows.get(workflow_name)
        if not workflow:
            raise ValueError(f"工作流未找到: {workflow_name}")
        
        if context is None:
            context = {}
        
        context['workflow'] = {
            'name': workflow.name,
            'start_time': datetime.now().isoformat(),
            'run_id': self._generate_run_id()
        }
        
        # 设置环境变量
        for key, value in workflow.env.items():
            context[f'env.{key}'] = value
        
        # 拓扑排序确定执行顺序
        execution_order = self._topological_sort(workflow.tasks)
        
        # 执行任务
        for task_id in execution_order:
            task = workflow.tasks[task_id]
            
            # 检查依赖是否满足
            if not self._dependencies_met(task, workflow.tasks):
                task.status = TaskStatus.SKIPPED
                continue
            
            # 执行任务
            await self._execute_task(task, context, workflow)
            
            # 检查是否失败
            if task.status == TaskStatus.FAILED:
                await self._handle_error(workflow, task, context)
                break
        
        return {
            'workflow': workflow.name,
            'status': 'success' if all(
                t.status == TaskStatus.SUCCESS for t in workflow.tasks.values()
            ) else 'failed',
            'tasks': {
                task_id: {
                    'status': task.status.value,
                    'output': task.output,
                    'error': task.error
                }
                for task_id, task in workflow.tasks.items()
            }
        }
    
    async def _execute_task(self, task: Task, context: Dict, workflow: Workflow):
        """执行单个任务"""
        task.status = TaskStatus.RUNNING
        task.started_at = datetime.now()
        
        handler = self.task_handlers.get(task.type)
        if not handler:
            task.status = TaskStatus.FAILED
            task.error = f"未知的任务类型: {task.type}"
            return
        
        # 模板变量替换
        config = self._render_template(task.config, context)
        
        # 重试逻辑
        for attempt in range(task.retries + 1):
            try:
                result = await handler(config, context)
                task.output = result
                task.status = TaskStatus.SUCCESS
                
                # 更新上下文
                context[f'tasks.{task.id}.output'] = result
                break
            except Exception as e:
                if attempt < task.retries:
                    await asyncio.sleep(2 ** attempt)  # 指数退避
                else:
                    task.status = TaskStatus.FAILED
                    task.error = str(e)
        
        task.completed_at = datetime.now()
    
    async def _handle_web_fetch(self, config: Dict, context: Dict) -> Any:
        """处理web_fetch任务"""
        from kimi_fetch import kimi_fetch
        
        url = config.get('url')
        if not url:
            raise ValueError("URL is required")
        
        result = kimi_fetch(url=url)
        return result
    
    async def _handle_exec(self, config: Dict, context: Dict) -> Any:
        """处理exec任务"""
        import subprocess
        
        command = config.get('command')
        if not command:
            raise ValueError("Command is required")
        
        result = subprocess.run(
            command,
            shell=True,
            capture_output=True,
            text=True,
            timeout=config.get('timeout', 300)
        )
        
        if result.returncode != 0:
            raise RuntimeError(f"Command failed: {result.stderr}")
        
        return result.stdout
    
    async def _handle_message(self, config: Dict, context: Dict) -> Any:
        """处理message任务"""
        # 实际实现中调用 message 工具
        channel = config.get('channel')
        message = config.get('message')
        
        print(f"[MESSAGE to {channel}]: {message}")
        return {'sent': True, 'channel': channel}
    
    def _topological_sort(self, tasks: Dict[str, Task]) -> List[str]:
        """拓扑排序"""
        in_degree = {task_id: 0 for task_id in tasks}
        graph = {task_id: [] for task_id in tasks}
        
        for task_id, task in tasks.items():
            for dep in task.depends_on:
                if dep in tasks:
                    graph[dep].append(task_id)
                    in_degree[task_id] += 1
        
        queue = [tid for tid, deg in in_degree.items() if deg == 0]
        result = []
        
        while queue:
            current = queue.pop(0)
            result.append(current)
            
            for neighbor in graph[current]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
        
        if len(result) != len(tasks):
            raise ValueError("工作流中存在循环依赖")
        
        return result
    
    def _dependencies_met(self, task: Task, tasks: Dict[str, Task]) -> bool:
        """检查依赖是否满足"""
        for dep_id in task.depends_on:
            dep_task = tasks.get(dep_id)
            if not dep_task or dep_task.status != TaskStatus.SUCCESS:
                return False
        return True
    
    def _render_template(self, config: Dict, context: Dict) -> Dict:
        """模板变量替换"""
        import re
        
        def replace_vars(value):
            if isinstance(value, str):
                pattern = r'\{\{\s*(\S+)\s*\}\}'
                def replacer(match):
                    var_path = match.group(1)
                    keys = var_path.split('.')
                    result = context
                    for key in keys:
                        if isinstance(result, dict):
                            result = result.get(key, match.group(0))
                        else:
                            return match.group(0)
                    return str(result)
                return re.sub(pattern, replacer, value)
            elif isinstance(value, dict):
                return {k: replace_vars(v) for k, v in value.items()}
            elif isinstance(value, list):
                return [replace_vars(item) for item in value]
            return value
        
        return replace_vars(config)
    
    async def _handle_error(self, workflow: Workflow, failed_task: Task, context: Dict):
        """处理错误"""
        context['failed_task'] = {'name': failed_task.name}
        context['error'] = {'message': failed_task.error}
        
        for error_handler in workflow.on_error:
            handler_type = error_handler.get('type')
            handler_config = error_handler.get('config', {})
            
            if handler_type in self.task_handlers:
                try:
                    await self.task_handlers[handler_type](handler_config, context)
                except Exception as e:
                    print(f"错误处理器失败: {e}")
    
    def _generate_run_id(self) -> str:
        """生成运行ID"""
        import uuid
        return str(uuid.uuid4())[:8]
```

```python
#!/usr/bin/env python3
# workflows/code_review.py
"""
自动化代码审查流程

步骤：
1. 获取代码变更
2. 静态分析
3. 安全检查
4. 生成审查报告
5. 通知相关人员
"""

import asyncio
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class ReviewResult:
    file: str
    issues: List[Dict]
    score: float

class CodeReviewWorkflow:
    """代码审查工作流"""
    
    def __init__(self, repo_path: str, base_branch: str = "main"):
        self.repo_path = repo_path
        self.base_branch = base_branch
        self.results: List[ReviewResult] = []
    
    async def run(self) -> Dict:
        """运行完整审查流程"""
        print("🔍 开始代码审查流程...")
        
        # 步骤1: 获取变更文件
        changed_files = await self._get_changed_files()
        print(f"📁 发现 {len(changed_files)} 个变更文件")
        
        # 步骤2: 并行执行多种检查
        tasks = [
            self._run_linter(changed_files),
            self._run_security_scan(changed_files),
            self._run_complexity_analysis(changed_files),
        ]
        
        linter_results, security_results, complexity_results = await asyncio.gather(*tasks)
        
        # 步骤3: 汇总结果
        summary = self._summarize_results(
            linter_results,
            security_results,
            complexity_results
        )
        
        # 步骤4: 生成报告
        report = self._generate_report(summary)
        
        # 步骤5: 发送通知
        await self._send_notification(report)
        
        return summary
    
    async def _get_changed_files(self) -> List[str]:
        """获取变更文件列表"""
        import subprocess
        
        result = subprocess.run(
            ['git', 'diff', '--name-only', f'{self.base_branch}...HEAD'],
            cwd=self.repo_path,
            capture_output=True,
            text=True
        )
        
        return [
            f.strip() for f in result.stdout.split('\n')
            if f.strip() and not f.startswith('.')
        ]
    
    async def _run_linter(self, files: List[str]) -> Dict:
        """运行代码检查工具"""
        print("🔧 运行代码检查...")
        
        issues = []
        for file in files:
            if file.endswith('.py'):
                # Python代码检查
                result = await self._run_command(['flake8', file])
                if result.returncode != 0:
                    issues.extend(self._parse_flake8_output(result.stdout))
            elif file.endswith('.js'):
                # JavaScript代码检查
                result = await self._run_command(['eslint', file])
                if result.returncode != 0:
                    issues.extend(self._parse_eslint_output(result.stdout))
        
        return {'tool': 'linter', 'issues': issues}
    
    async def _run_security_scan(self, files: List[str]) -> Dict:
        """运行安全扫描"""
        print("🔒 运行安全扫描...")
        
        security_issues = []
        
        # 检查敏感信息泄露
        for file in files:
            content = await self._read_file(file)
            
            # 检查硬编码密钥
            if 'api_key' in content.lower() or 'password' in content.lower():
                security_issues.append({
                    'file': file,
                    'type': 'potential_secret',
                    'severity': 'high',
                    'message': 'Possible hardcoded secret detected'
                })
            
            # 检查SQL注入风险
            if 'execute(' in content and ('%' in content or '+' in content):
                security_issues.append({
                    'file': file,
                    'type': 'sql_injection',
                    'severity': 'critical',
                    'message': 'Potential SQL injection vulnerability'
                })
        
        return {'tool': 'security', 'issues': security_issues}
    
    async def _run_complexity_analysis(self, files: List[str]) -> Dict:
        """运行复杂度分析"""
        print("📊 运行复杂度分析...")
        
        complexity_issues = []
        
        for file in files:
            if file.endswith('.py'):
                result = await self._run_command(['radon', 'cc', '-a', file])
                complexity_issues.extend(self._parse_radon_output(result.stdout))
        
        return {'tool': 'complexity', 'issues': complexity_issues}
    
    async def _run_command(self, cmd: List[str]):
        """执行命令"""
        import subprocess
        
        proc = await asyncio.create_subprocess_exec(
            *cmd,
            cwd=self.repo_path,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )
        
        stdout, stderr = await proc.communicate()
        
        return subprocess.CompletedProcess(
            args=cmd,
            returncode=proc.returncode,
            stdout=stdout.decode(),
            stderr=stderr.decode()
        )
    
    async def _read_file(self, file: str) -> str:
        """读取文件内容"""
        import aiofiles
        
        try:
            async with aiofiles.open(f"{self.repo_path}/{file}", 'r') as f:
                return await f.read()
        except:
            return ''
    
    def _parse_flake8_output(self, output: str) -> List[Dict]:
        """解析flake8输出"""
        issues = []
        for line in output.split('\n'):
            if ':' in line:
                parts = line.split(':')
                if len(parts) >= 4:
                    issues.append({
                        'file': parts[0],
                        'line': int(parts[1]),
                        'column': int(parts[2]),
                        'message': ':'.join(parts[3:]),
                        'severity': 'warning'
                    })
        return issues
    
    def _parse_eslint_output(self, output: str) -> List[Dict]:
        """解析eslint输出"""
        # 简化实现
        return []
    
    def _parse_radon_output(self, output: str) -> List[Dict]:
        """解析radon输出"""
        issues = []
        for line in output.split('\n'):
            if 'ERROR' in line or 'WARNING' in line:
                issues.append({
                    'message': line,
                    'severity': 'info'
                })
        return issues
    
    def _summarize_results(self, *results) -> Dict:
        """汇总结果"""
        all_issues = []
        for result in results:
            all_issues.extend(result.get('issues', []))
        
        # 按严重程度分组
        severity_counts = {'critical': 0, 'high': 0, 'medium': 0, 'low': 0, 'warning': 0, 'info': 0}
        for issue in all_issues:
            severity = issue.get('severity', 'info')
            severity_counts[severity] = severity_counts.get(severity, 0) + 1
        
        # 计算总分 (满分100)
        score = 100
        score -= severity_counts['critical'] * 20
        score -= severity_counts['high'] * 10
        score -= severity_counts['medium'] * 5
        score -= severity_counts['low'] * 2
        score -= severity_counts['warning'] * 1
        score = max(0, score)
        
        return {
            'total_issues': len(all_issues),
            'severity_counts': severity_counts,
            'score': score,
            'passed': score >= 80 and severity_counts['critical'] == 0,
            'issues': all_issues
        }
    
    def _generate_report(self, summary: Dict) -> str:
        """生成审查报告"""
        lines = [
            "# 代码审查报告",
            "",
            f"**总分**: {summary['score']}/100",
            f"**状态**: {'✅ 通过' if summary['passed'] else '❌ 未通过'}",
            "",
            "## 问题统计",
            f"- 严重: {summary['severity_counts']['critical']}",
            f"- 高: {summary['severity_counts']['high']}",
            f"- 中: {summary['severity_counts']['medium']}",
            f"- 低: {summary['severity_counts']['low']}",
            f"- 警告: {summary['severity_counts']['warning']}",
            "",
            "## 详细问题",
        ]
        
        for issue in summary['issues'][:20]:  # 只显示前20个
            lines.append(f"- [{issue.get('severity', 'info').upper()}] {issue.get('file', 'N/A')}: {issue.get('message', '')}")
        
        if len(summary['issues']) > 20:
            lines.append(f"\n... 还有 {len(summary['issues']) - 20} 个问题")
        
        return '\n'.join(lines)
    
    async def _send_notification(self, report: str):
        """发送通知"""
        print("📧 发送审查报告...")
        # 实际实现中调用 message 工具
        print(report)
```

```python
#!/usr/bin/env python3
# scripts/monitor_alerts.py
"""
OpenClaw 监控告警系统
"""

import asyncio
import json
import smtplib
from datetime import datetime, timedelta
from typing import Dict, List, Optional
from dataclasses import dataclass
from email.mime.text import MIMEText
import aiohttp

@dataclass
class AlertRule:
    """告警规则"""
    name: str
    condition: str  # Python表达式
    severity: str  # critical, warning, info
    cooldown_minutes: int = 60
    last_triggered: Optional[datetime] = None

@dataclass
class Alert:
    """告警"""
    rule_name: str
    severity: str
    message: str
    timestamp: datetime
    metrics: Dict

class AlertManager:
    """告警管理器"""
    
    def __init__(self, config_path: str = "config/alerts.yaml"):
        self.config = self._load_config(config_path)
        self.rules: List[AlertRule] = []
        self.alert_history: List[Alert] = []
        self._load_rules()
    
    def _load_config(self, path: str) -> Dict:
        import yaml
        try:
            with open(path, 'r') as f:
                return yaml.safe_load(f)
        except FileNotFoundError:
            return self._default_config()
    
    def _default_config(self) -> Dict:
        return {
            'notification': {
                'channels': ['console'],
                'email': {
                    'smtp_host': 'smtp.gmail.com',
                    'smtp_port': 587,
                    'from': 'alerts@example.com',
                    'to': ['admin@example.com']
                }
            }
        }
    
    def _load_rules(self):
        """加载告警规则"""
        default_rules = [
            AlertRule(
                name="high_cpu",
                condition="metrics['cpu_percent'] > 80",
                severity="warning",
                cooldown_minutes=30
            ),
            AlertRule(
                name="critical_cpu",
                condition="metrics['cpu_percent'] > 95",
                severity="critical",
                cooldown_minutes=15
            ),
            AlertRule(
                name="high_memory",
                condition="metrics['memory_percent'] > 85",
                severity="warning",
                cooldown_minutes=30
            ),
            AlertRule(
                name="disk_full",
                condition="metrics['disk_percent'] > 90",
                severity="critical",
                cooldown_minutes=60
            ),
            AlertRule(
                name="service_down",
                condition="metrics['gateway_status'] != 'running'",
                severity="critical",
                cooldown_minutes=5
            ),
            AlertRule(
                name="high_error_rate",
                condition="metrics['error_rate_5m'] > 0.1",
                severity="warning",
                cooldown_minutes=30
            )
        ]
        self.rules = default_rules
    
    async def collect_metrics(self) -> Dict:
        """收集指标"""
        import psutil
        
        # 系统指标
        cpu_percent = psutil.cpu_percent(interval=1)
        memory = psutil.virtual_memory()
        disk = psutil.disk_usage('/')
        
        # 进程指标
        try:
            proc = psutil.Process()
            proc_cpu = proc.cpu_percent()
            proc_memory = proc.memory_info().rss / 1024 / 1024  # MB
        except:
            proc_cpu = 0
            proc_memory = 0
        
        # 服务状态
        gateway_status = await self._check_gateway_status()
        
        # 错误率（从日志统计）
        error_rate = await self._calculate_error_rate(minutes=5)
        
        return {
            'timestamp': datetime.now().isoformat(),
            'cpu_percent': cpu_percent,
            'memory_percent': memory.percent,
            'memory_used_mb': memory.used / 1024 / 1024,
            'disk_percent': disk.percent,
            'proc_cpu_percent': proc_cpu,
            'proc_memory_mb': proc_memory,
            'gateway_status': gateway_status,
            'error_rate_5m': error_rate,
            'open_files': len(proc.open_files()) if hasattr(proc, 'open_files') else 0,
            'connections': len(proc.connections()) if hasattr(proc, 'connections') else 0
        }
    
    async def _check_gateway_status(self) -> str:
        """检查Gateway状态"""
        import subprocess
        try:
            result = subprocess.run(
                ['openclaw', 'gateway', 'status'],
                capture_output=True,
                timeout=5
            )
            return 'running' if result.returncode == 0 else 'stopped'
        except:
            return 'unknown'
    
    async def _calculate_error_rate(self, minutes: int = 5) -> float:
        """计算错误率"""
        import subprocess
        from pathlib import Path
        
        log_file = Path.home() / '.openclaw' / 'logs' / 'openclaw.log'
        if not log_file.exists():
            return 0.0
        
        try:
            # 统计最近N分钟的ERROR日志
            since_time = datetime.now() - timedelta(minutes=minutes)
            result = subprocess.run(
                ['grep', '-c', 'ERROR', str(log_file)],
                capture_output=True,
                text=True
            )
            error_count = int(result.stdout.strip() or 0)
            
            # 简单估算：假设每分钟平均100条日志
            total_estimate = minutes * 100
            return min(error_count / total_estimate, 1.0) if total_estimate > 0 else 0.0
        except:
            return 0.0
    
    def evaluate_rules(self, metrics: Dict) -> List[Alert]:
        """评估告警规则"""
        alerts = []
        now = datetime.now()
        
        for rule in self.rules:
            # 检查冷却期
            if rule.last_triggered:
                cooldown = timedelta(minutes=rule.cooldown_minutes)
                if now - rule.last_triggered < cooldown:
                    continue
            
            # 评估条件
            try:
                if eval(rule.condition, {'metrics': metrics}):
                    alert = Alert(
                        rule_name=rule.name,
                        severity=rule.severity,
                        message=f"Rule '{rule.name}' triggered: {rule.condition}",
                        timestamp=now,
                        metrics=metrics.copy()
                    )
                    alerts.append(alert)
                    rule.last_triggered = now
            except Exception as e:
                print(f"Error evaluating rule {rule.name}: {e}")
        
        return alerts
    
    async def send_notifications(self, alerts: List[Alert]):
        """发送告警通知"""
        if not alerts:
            return
        
        channels = self.config.get('notification', {}).get('channels', ['console'])
        
        for channel in channels:
            if channel == 'console':
                self._notify_console(alerts)
            elif channel == 'email':
                await self._notify_email(alerts)
            elif channel == 'slack':
                await self._notify_slack(alerts)
    
    def _notify_console(self, alerts: List[Alert]):
        """控制台通知"""
        for alert in alerts:
            severity_colors = {
                'critical': '\033[91m',  # Red
                'warning': '\033[93m',   # Yellow
                'info': '\033[94m'       # Blue
            }
            color = severity_colors.get(alert.severity, '')
            reset = '\033[0m'
            
            print(f"{color}[ALERT {alert.severity.upper()}] {alert.rule_name}{reset}")
            print(f"  Time: {alert.timestamp}")
            print(f"  Message: {alert.message}")
            print(f"  Metrics: {json.dumps(alert.metrics, indent=2)}")
            print()
    
    async def _notify_email(self, alerts: List[Alert]):
        """邮件通知"""
        email_config = self.config.get('notification', {}).get('email', {})
        if not email_config:
            return
        
        subject = f"OpenClaw Alert: {len(alerts)} alerts triggered"
        body = "\n\n".join([
            f"[{a.severity.upper()}] {a.rule_name}\n{a.message}\nTime: {a.timestamp}"
            for a in alerts
        ])
        
        msg = MIMEText(body)
        msg['Subject'] = subject
        msg['From'] = email_config.get('from')
        msg['To'] = ', '.join(email_config.get('to', []))
        
        try:
            with smtplib.SMTP(email_config.get('smtp_host'), email_config.get('smtp_port')) as server:
                server.starttls()
                # server.login(email_config.get('username'), email_config.get('password'))
                server.send_message(msg)
        except Exception as e:
            print(f"Failed to send email: {e}")
    
    async def _notify_slack(self, alerts: List[Alert]):
        """Slack通知"""
        slack_config = self.config.get('notification', {}).get('slack', {})
        webhook_url = slack_config.get('webhook_url')
        
        if not webhook_url:
            return
        
        for alert in alerts:
            color = {
                'critical': 'danger',
                'warning': 'warning',
                'info': 'good'
            }.get(alert.severity, 'warning')
            
            payload = {
                'attachments': [{
                    'color': color,
                    'title': f"OpenClaw Alert: {alert.rule_name}",
                    'text': alert.message,
                    'fields': [
                        {'title': 'Severity', 'value': alert.severity, 'short': True},
                        {'title': 'Time', 'value': str(alert.timestamp), 'short': True}
                    ],
                    'footer': 'OpenClaw Monitoring'
                }]
            }
            
            try:
                async with aiohttp.ClientSession() as session:
                    async with session.post(webhook_url, json=payload) as resp:
                        if resp.status != 200:
                            print(f"Failed to send Slack notification: {resp.status}")
            except Exception as e:
                print(f"Failed to send Slack notification: {e}")
    
    async def run(self, interval_seconds: int = 60):
        """运行监控循环"""
        print(f"Starting monitoring with {len(self.rules)} rules...")
        
        while True:
            try:
                # 收集指标
                metrics = await self.collect_metrics()
                print(f"[{datetime.now()}] Metrics collected")
                
                # 评估规则
                alerts = self.evaluate_rules(metrics)
                
                if alerts:
                    print(f"{len(alerts)} alerts triggered")
                    self.alert_history.extend(alerts)
                    
                    # 发送通知
                    await self.send_notifications(alerts)
                
                # 保存指标历史
                await self._save_metrics(metrics)
                
            except Exception as e:
                print(f"Error in monitoring loop: {e}")
            
            await asyncio.sleep(interval_seconds)
    
    async def _save_metrics(self, metrics: Dict):
        """保存指标到文件"""
        from pathlib import Path
        import json
        
        metrics_dir = Path.home() / '.openclaw' / 'metrics'
        metrics_dir.mkdir(parents=True, exist_ok=True)
        
        date_str = datetime.now().strftime('%Y%m%d')
        file_path = metrics_dir / f'metrics-{date_str}.jsonl'
        
        with open(file_path, 'a') as f:
            f.write(json.dumps(metrics) + '\n')

# 使用示例
async def main():
    manager = AlertManager()
    await manager.run(interval_seconds=30)

if __name__ == '__main__':
    asyncio.run(main())
```
